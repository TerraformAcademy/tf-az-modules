## How to use

 ``` hcl
 module "agw-XXXX" {
  source = "./tf-azurerm-modules/networking/azurerm-nsg"

  name                = "test-name"
  resource_group_name = "test-resource_group_name"
  location            = "centralus"
  tags                = %TAGGING_MODULE_OUTPUT%
  sku  = {
    name     = "Standard_v2"
    tier     = "Standard_v2"
    capacity = 2
  }

  backend_address_pool = [ {
    name = local.backend_address_pool_name
  } ]

  backend_http_settings = [ {
    name                  = local.http_setting_name
    cookie_based_affinity = "Disabled"
    path                  = "/path1/"
    port                  = 80
    protocol              = "Http"
    request_timeout       = 60
  }]
}
```
